function calculateTotal(items, taxRate = 0.08) {
  const subtotal = items.reduce((sum, { price, quantity }) =>
    sum + (price * quantity), 0);
  const tax = subtotal * taxRate;
  return { subtotal, tax, total: subtotal + tax };
}
